flowchart TD
    start([Start]) --> canvas[Access Canvas dropbox assessment]
    canvas --> takePicture[Select option to take picture]
    takePicture --> holdPaper[Hold paper in front of camera]
    holdPaper --> capture[Camera captures assignment]
    capture --> multiPage{Multiple pages?}
    
    multiPage -->|Yes| scanMore[Select option to scan more pages]
    scanMore --> holdPaper
    
    multiPage -->|No| review[Review captured assignment]
    review --> satisfied{Satisfied with capture?}
    satisfied -->|No| holdPaper
    
    satisfied -->|Yes| submit[Submit assignment to teacher]
    submit --> end([End])
